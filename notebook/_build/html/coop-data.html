import pandas as pd
import altair as alt

sales = pd.read_csv("./seward_sales_since_2020.csv",thousands=',')
sales.sale_date = pd.to_datetime(sales.sale_date)
source = sales

All Sales (2020-2024)#

The chart below includes all recorded sales (transfers excluded) between 2020 and 2024 (through June).

  • Use dropdowns to filter by building and unit number/layout.

  • Hover on points to see address, unit number, floor, # bedrooms, # bathrooms, sale price, and sale date.

  • Clicking on points in the chart will link to Streeteasy listings where available (Ctrl + click to open in another tab)

  • Click on the bedroom count in the legend will de-emphaize other bedroom data.

base = alt.Chart(source).mark_line(point=alt.OverlayMarkDef(filled=False, fill="white", size=50.0)).encode(
    alt.X('sale_date:T', axis=alt.Axis(format="%Y %B")),
    alt.Y('sale_price:Q'),
    alt.Color(
        'br:N',
        sort=color_order
    ).scale(),
    href='url:N',
    opacity=alt.condition(selection, alt.value(1), alt.value(0.1)),
    tooltip = ['address:N','unit:N','floor:N','br:N','ba:N','sale_price:Q', 'sale_date:T']
).add_params(
    selection
).properties(
    width=600,
    height=350,
    title="Sale Prices by # Bedrooms (2020-2024)"
).interactive()

address_dropdown = alt.binding_select(options=addresses, name="Address")
address_select = alt.selection_point(fields=['address'], bind=address_dropdown)
unit_dropdown = alt.binding_select(options=unit_number, name="Unit Number")
unit_select = alt.selection_point(fields=['number'], bind=unit_dropdown)

filters = base.add_params(
    unit_select
).transform_filter(
    unit_select
).add_params(
    address_select

Median Sale Price by # Bedrooms and Floor (2020-2024)#

The charts below show median sale price by number of bedrooms and floor in two views.

  • Click the legend to filter by floor in the first chart and bedroom count in the second.

  • Hover on points to see median sale price and number of sales included in the aggregate.

Show code cell source Hide code cell source 
selection_1 = alt.selection_point(fields=['floor'], bind='legend')
x_order = ["studio", "1", "2", "3", "4"]

floor = alt.Chart(source).mark_point().encode(
    alt.X(
        'br:N',
        scale=alt.Scale(domain=x_order),
        sort=x_order,
    ),
    alt.Y('median(sale_price):Q',scale=alt.Scale(domain=(0, 3_000_000))),
    alt.Color('floor:N', scale=alt.Scale(domain=list(range(1,21)))),
    opacity=alt.condition(selection_1, alt.value(1), alt.value(0.1)),
    size=alt.condition(selection_1, alt.value(100), alt.value(30)),
    tooltip = ['median(sale_price):Q', 'count(sale_price):Q']
).add_params(
    selection_1
).properties(
    width=600,
    height=350,
    title="Median Sale Price by # Bedrooms and Floor (2020-2024)"
).transform_filter(
    selection_1
).interactive()

selection_2 = alt.selection_point(fields=['br'], bind='legend')

br = alt.Chart(source).mark_line(point=alt.OverlayMarkDef(filled=False, fill="white", size=50.0)).encode(
    alt.X(
        'floor:N',
    ),
    alt.Y('median(sale_price):Q'),
    alt.Color(
        'br:N',
        sort=x_order
    ).scale(),
    opacity=alt.condition(selection_2, alt.value(1), alt.value(0.1)),
    tooltip = ['median(sale_price):Q', 'count(sale_price):Q']
).add_params(
    selection_2
).properties(
    width=600,
    height=350,
    title="Median Sale Price by # Bedrooms and Floor (2020-2024)"
).interactive()

alt.vconcat(br, floor).resolve_scale(
    color='independent'
)
